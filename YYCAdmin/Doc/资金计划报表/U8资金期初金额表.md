# U8资金期初金额表 开发规范与同步接口文档

## 一、业务背景与目标
为实现用友 U8 系统中的资金数据能够作为 YS (YonSuite / YonBIP) 自定义报表（如“金桥资金计划数据报表”）的基础期初数据，需要建立一个自定义实体（档案）用于存储“U8资金期初金额”。
本接口支持外部系统（U8 或中间件）按月批量推送各组织的期初/期末资金数据，并在 YS 侧进行自动的“增量创建或更新”处理。

---

## 二、自定义实体（实体编码：AT23E0C31009D80007.F25303_TOBalance.U8TOBalance）
在 YonBuilder 中创建的自定义实体设计如下：

### 1. 字段设计表
| 字段中文名称 | 字段编码 (Code) | 字段类型 | 约束条件 | 说明 |
| :--- | :--- | :--- | :--- | :--- |
| 主键ID | id | 字符串 (String) | 主键，自动生成 | 唯一主键 |
| 组织 | org_id | 参照 (Association) | 必填 | **直接关联 YS 标准组织档案实体（`org.func.BaseOrg`），存储组织ID** |
| 日期 | cDate | 日期 (Date) | 必填 | 格式为 `YYYY-MM-DD`，业务规则限定必须为当月 1 号（期初） |
| 锁定金额期初 | OLAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 锁定金额期初数 |
| 锁定金额期末 | CLAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 锁定金额期末数 |
| 可支配金额期初 | OAAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 可支配金额期初数 |
| 可支配金额期末 | CAAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 可支配金额期末数 |
| 汇票锁定金额期初 | HOLAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 汇票锁定金额期初数 |
| 汇票锁定金额期末 | HCLAmount | 数值 (Decimal 28, 2) | 可空，默认为 0.00 | 汇票锁定金额期末数 |
| 是否启用 | enable | 布尔 (Boolean) | 默认 true | 控制该条期初记录是否参与计算，**接口不传入，依靠数据库默认值 true 或后台自动启用** |

### 2. 唯一性校验规则
* **联合唯一约束**：`org_id` (组织ID) + `cDate` (日期)
* **业务逻辑**：每个组织在每个自然月内**仅允许存在一条期初数据记录**。

## 三、同步接口规范 (API Specification)
接口用于接收 U8 推送的数据并持久化到 YS 自定义实体中。

* **请求路径**：`/api/v1/custom/u8/capital/beginning/sync` *(在 openapi 中注册的自定义API路径)*
* **请求方法**：`POST`
* **Content-Type**：`application/json`

### 1. 请求体 JSON 示例
*注：`enable` 字段不需要接口传入，默认在实体中自动生效。组织编码属性在请求体中传参为 `code`。*

```json
    {
```

```json

  "code": "200",
```

## 四、YonScript 核心逻辑设计 (API 脚本模板)
在 YS 自定义 API 的脚本编辑器中，处理逻辑实现如下（采用外部批量写操作，符合平台静态校验性能规范）：

```javascript
let AbstractAPIHandler = require('AbstractAPIHandler');

class MyAPIHandler extends AbstractAPIHandler {
  execute(request) {
    // 基础辅助解析与防 SQL 注入函数
    function parseRequest(raw) {
      if (!raw) return {};
      if (typeof raw === "string") {
        try {
          return JSON.parse(raw);
        } catch (e) {
          throw new Error("请求体不是合法 JSON。");
        }
      }
      return raw;
    }

    function esc(v) {
      return String(v == null ? "" : v).replace(/'/g, "''");
    }

    function toNum(v) {
      let n = Number(v);
      return isNaN(n) ? 0 : n;
    }

    function query(sql, domainKey) {
      if (domainKey) return ObjectStore.queryByYonQL(sql, domainKey) || [];
      return ObjectStore.queryByYonQL(sql) || [];
    }

    let params = parseRequest(request);
    let dataList = params.Data;
    if (!dataList || !Array.isArray(dataList)) {
      return { code: "400", message: "请求体为空或缺失 Data 节点", success: false };
    }
    
    let successCount = 0;
    let updateCount = 0;
    let insertCount = 0;
    
    const ENTITY_URI = "AT23E0C31009D80007.F25303_TOBalance.U8TOBalance";
    
    let insertList = [];
    let updateList = [];
    
    // 遍历数据包
    for (let i = 0; i < dataList.length; i++) {
      let item = dataList[i];
      let orgCode = String(item.code || "").trim(); // 使用 code 作为组织编码
      let cDate = String(item.cDate || "").trim();
      
      // 1. 参数合理性基础校验
      if (!orgCode || !cDate) {
        throw new Error("第 " + (i + 1) + " 条数据校验错误：组织编码(code)或日期(cDate)不能为空");
      }
      
      // 强制校验日期必须为当月 1 号（期初格式）
      if (!cDate.endsWith("-01")) {
        throw new Error("第 " + (i + 1) + " 条数据校验错误：日期格式必须为每月1号，如 YYYY-MM-01");
      }

      // 2. 根据组织编码，直接检索 YS 标准组织档案（org.func.BaseOrg）获取 org_id
      // 采用系统推荐的 domainKey 'org' 与实体进行查询
      let orgQuerySql = `select id, name from org.func.BaseOrg where orgtype = '1' and code = '${esc(orgCode)}'`;
      let orgRes = query(orgQuerySql, "org");
      
      if (!orgRes || orgRes.length === 0) {
        throw new Error("同步失败：第 " + (i + 1) + " 条记录中组织编码 [" + orgCode + "] 在 YS 组织档案中不存在或类型非行政组织");
      }
      let orgId = orgRes[0].id;

      // 3. 检查当前 组织 + 日期 记录是否已在 YS 存在
      let existQuerySql = `select id from ${ENTITY_URI} where org_id = '${esc(orgId)}' and cDate = '${esc(cDate)}'`;
      let existRes = query(existQuerySql);
      
      let targetRecord = {
        org_id: orgId,
        cDate: cDate,
        OLAmount: toNum(item.OLAmount),
        CLAmount: toNum(item.CLAmount),
        OAAmount: toNum(item.OAAmount),
        CAAmount: toNum(item.CAAmount),
        HOLAmount: toNum(item.HOLAmount),
        HCLAmount: toNum(item.HCLAmount)
      };

      if (existRes && existRes.length > 0) {
        // 4. 已存在：收集主键 ID，加入批量更新队列
        targetRecord.id = existRes[0].id;
        updateList.push(targetRecord);
        updateCount++;
      } else {
        // 5. 不存在：启用标识置为 true，加入批量新增队列
        targetRecord.enable = true;
        insertList.push(targetRecord);
        insertCount++;
      }
      successCount++;
    }
    
    // 6. 平台性能合规优化：在循环体外执行批量数据库写入操作（防止 N+1 写操作编译报错）
    if (insertList.length > 0) {
      ObjectStore.insertBatch(ENTITY_URI, insertList);
    }
    if (updateList.length > 0) {
      ObjectStore.updateBatch(ENTITY_URI, updateList);
    }
    
    return {
      code: "200",
      message: "同步成功：共处理 " + successCount + " 条数据（新增 " + insertCount + " 条，更新 " + updateCount + " 条）。",
      success: true
    };
  }
}

exports({ entryPoint: MyAPIHandler });
```
